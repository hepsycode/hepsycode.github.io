67 void mainsystem::fir8_shifting()
68 {
69  // datatype for channels
70  fir8s_parameters fir8s_p;
71  fir8s_results fir8s_r;
72 
73  // local variables
74  sc_uint<8> sample_tmp;
75  sc_uint<8> shift[8];
76 
77  HEPSY_S(fir8s_id) while(1)
78  {HEPSY_S(fir8s_id)
79 
80  // read parameters from channel
81  HEPSY_S(fir8s_id) fir8s_p=fir8s_parameters_channel->read();
82 
83  // fill local variables
84  HEPSY_S(fir8s_id) sample_tmp=fir8s_p.sample_tmp;
85 
86  HEPSY_S(fir8s_id) for (unsigned j = 0; j < TAP8; j++)
87  {HEPSY_S(fir8s_id)
88  HEPSY_S(fir8s_id) shift[j] = fir8s_p.shift[j];
89  HEPSY_S(fir8s_id)}
90 
91  // process
92 
93  HEPSY_S(fir8s_id) for(int i=TAP8-2; i>=0; i--)
94  {HEPSY_S(fir8s_id)
95  HEPSY_S(fir8s_id) shift[i+1] = shift[i];
96  HEPSY_S(fir8s_id)}
97 
98  HEPSY_S(fir8s_id) shift[0]=sample_tmp;
99 
100  // fill datatype
101  HEPSY_S(fir8s_id) for( unsigned j=0; j<TAP8; j++) fir8s_p.shift[j]=shift[j];
102 
103  // send results by channel
104  HEPSY_S(fir8s_id) fir8s_results_channel->write(fir8s_r);
105 
106  // Profiling
107  HEPSY_P(fir8s_id)
108  }
109 }
