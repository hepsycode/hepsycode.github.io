68 {
69  // datatype for channels
70  fir8e_parameters fir8e_p;
71  fir8e_results fir8e_r;
72 
73  // local variables
74  sc_int<17> pro;
75  sc_uint<19> acc;
76  sc_uint<9> coef[8];
77  sc_uint<8> sample_tmp;
78  sc_uint<8> shift[8];
79 
80  HEPSY_S(fir8e_id) while(1) //HEPSY_S rules for WHILE: before while, and after {
81  {HEPSY_S(fir8e_id)
82 
83  // read parameters from channel
84  HEPSY_S(fir8e_id) fir8e_p=fir8e_parameters_channel->read();
85 
86  // fill local variables
87  HEPSY_S(fir8e_id) sample_tmp=fir8e_p.sample_tmp;
88 
89  HEPSY_S(fir8e_id) for (unsigned j = 0; j < TAP8; j++)
90  {HEPSY_S(fir8e_id)
91  HEPSY_S(fir8e_id)coef[j] = fir8e_p.coef[j];
92  HEPSY_S(fir8e_id)}
93 
94  HEPSY_S(fir8e_id) for (unsigned j = 0; j < TAP8; j++)
95  {HEPSY_S(fir8e_id)
96  HEPSY_S(fir8e_id) shift[j] = fir8e_p.shift[j];
97  HEPSY_S(fir8e_id)}
98 
99  // process
100  HEPSY_S(fir8e_id) acc=sample_tmp*coef[0];
101 
102  HEPSY_S(fir8e_id) for(int i=TAP8-2; i>=0; i--) //HEPSY_S rules for FOR: before for, after {, and first of }
103  {HEPSY_S(fir8e_id)
104  HEPSY_S(fir8e_id) pro=shift[i]*coef[i+1];
105  HEPSY_S(fir8e_id) acc += pro;
106  HEPSY_S(fir8e_id)}
107 
108  // fill datatype
109  HEPSY_S(fir8e_id) fir8e_r.acc=acc;
110 
111  // send results by channel
112  HEPSY_S(fir8e_id) fir8e_results_channel->write(fir8e_r);
113 
114  // Profiling
115  HEPSY_P(fir8e_id)
116  }
117 }
